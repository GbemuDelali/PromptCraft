OpenPrompt Examples

Example 1: Simple Prompt

        
# Simple Prompt Example
prompt("Generate a creative story about a dragon and a knight.")

Example 2: Using a Template

        
# Using a Template Example
template introduction {
  "Hello, <<name>>! Welcome to OpenPrompt. How can I assist you today?"
}

prompt(introduction, { "name": "User" })

Example 3: Parameterized Prompt

      
        # Parameterized Prompt Example
        template favoriteColor {
          "What is your favorite color, <>?"
        }
        
        prompt(favoriteColor, { "name": "Alice" })

Example 4: Dynamic Content Insertion 

      
        # Dynamic Content Insertion Example
        template personalizedMessage {
          "Hello, <>! Your favorite animal is a <>."
        }
        
        prompt(personalizedMessage, { "name": "Bob", "animal": "lion" })
